================================================================================
Including a flat class
================================================================================

class Foo
  include Bar
end

--------------------------------------------------------------------------------

(program
  (declaration
    (class_declaration
      (class_name)
      (members
        (member
          (include_member
            (class_name)))))))

================================================================================
Including a flat class with type arguments
================================================================================

end

--------------------------------------------------------------------------------

(program
  (declaration
    (class_declaration
      (class_name)
      (members
        (member
          (include_member
            (class_name)
            (type_arguments
              (type
                (namespaceable_type
                  (class_name))))))))))
